\matrix (A) [collection] {
    \node [elements before]; &
    \node (an-i) {$a_{n - i}$}; &
    \node [elements between]; &
    \node (an-1) {$a_{n - 1}$}; &
    \node (an) {$a_n$}; \\
};

\foreach \i/\r in {n/\neq,n-1/\neq,n-i/=}{
    \draw [subflow ->] (a\i.north) -- +(0, .5)
        node (c) [subelement, above] {$\texttt{a} \r a_{\i}$};
}

\draw [<- subflow, iterate ->] (an.north) arc;
\draw [subflow ->, <- iterate] (an.north) to (an-1.north);
\draw [subflow ->, <- iterate, dashed] (an-1.north) arc;
\draw [<- subflow, iterate ->, dashed] (an-i.north) arc;

    node (b) [element, below] {$n - i$};
