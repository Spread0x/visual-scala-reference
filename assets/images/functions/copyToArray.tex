\matrix (A) [collection] {
    \node (a1) {$a_1$}; &
    \node (a2) {$a_2$}; &
    \node [elements between]; &
    \node (an) {$a_n$}; \\
};

\matrix (B) [collection, below=2 of a1.south west, anchor=north west] {
    \node (b1) {$a_1$}; &
    \node (b2) {$a_2$}; &
    \node [elements between]; &
    \node (bn) {$a_n$}; &
    \node (bn+1) {$b_{n + 1}$}; &
    \node [elements after]; \\
};

\braceflow[-.2]{(a1.south west)}{(an.south east)}
\coordinate (x) at (lastbrace);
\braceflow[.2]{(b1.north west)}{(bn.north east)}
\draw [flow ->, out=270, in=90] (x) to (lastbrace);

\measure[-.5]{$n$}{b1.south west}{bn.south east}
\draw [flow ->] (last measure) -- +(0, -1)
    node [element, below] {$n$};
