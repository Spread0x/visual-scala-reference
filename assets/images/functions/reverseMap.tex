    \node {$a_2$}; &
    \node [elements between]; &

    \node [elements between]; &
    \node {$a_n$}; \\

    \node [elements between=.75]; &

    \node (f) [subfunction, anchor=north io 1] at ($ (a\i.south) + (0, -.5) $) {$\texttt{f}$};

    \draw [subflow ->] (f.south io 1) .. controls +(0, -.5) and +(0, .5) .. (b\i.north);
