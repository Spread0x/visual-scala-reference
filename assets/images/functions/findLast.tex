\matrix (A) [collection] {
    \node [elements before]; &
    \node (ai) {$a_i$}; &
    \node [elements between]; &
    \node (an-1) {$a_{n - 1}$}; &
    \node (an) {$a_n$}; \\
};

\foreach \i/\r in {n/\false,n-1/\false,i/\true}{
    \draw [subflow ->] (a\i.north) -- +(0, .5)
        node (p) [subfunction, above] {\texttt{p}};
    \draw [subflow ->] (p.north io 1) -- +(0, .5)
        node [subelement, above] {\r};
}

\draw [flow ->] (ai.south) -- +(0, -1)
    node [some, below] {$a_i$};

\draw [<- subflow] (an.north) arc [iterate ->];
\draw [subflow ->, <- iterate] (an.north) to (an-1.north);
\draw [subflow ->, <- iterate, dashed] (an-1.north) arc;
\draw [<- subflow, iterate ->, dashed] (ai.north) arc;
